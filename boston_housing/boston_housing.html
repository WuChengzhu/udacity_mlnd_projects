def performance_metric2(y_true, y_predict):
    """"""
    ss_res = sum([(yt-yp)**2 for yt,yp in zip(y_true, y_predict)])
    ss_tot = sum([(yt-np.mean(y_true))**2 for yt in y_true])
    score = 1 - (ss_res / ss_tot)

    return score

from sklearn.model_selection import KFold, GridSearchCV
from sklearn.tree import DecisionTreeRegressor
from sklearn.metrics import make_scorer


def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold()
    
    regressor = DecisionTreeRegressor()

    params = dict(max_depth = np.arange(1,11))

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, params, scoring=scoring_fnc, cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)

    # 
#     print pd.DataFrame(grid.cv_results_)
    return grid.best_estimator_

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    scores = [0, 0, 0]
    XS = np.array_split(X, 3)  # X3
    ys = np.array_split(y, 3)  # y3
    for i in range(3):
        x_val = XS[i]  # x
        y_val = ys[i]  # y
        x_tra = pd.concat([a for j,a in enumerate(XS) if j != i])  # x
        y_tra = pd.concat([a for j,a in enumerate(ys) if j != i])  # y
        score = scoring(y_val, estimator.fit(x_tra, y_tra).predict(x_val))  # 
        scores[i] = score
    return scores

def fit_model2(X, y):
    """  [X,y]"""

    model_scores = []
    for i in range(1, 11):
        model_score = np.mean(cross_val_score(DecisionTreeRegressor(max_depth=i), X, y))
        model_scores.append(model_score)
    best_max_depth = model_scores.index(max(model_scores))
    best_estimator = DecisionTreeRegressor(max_depth=best_max_depth)
